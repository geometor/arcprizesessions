import numpy as np

def apply_transformation(grid):
  """Applies the transformation rule to the input grid."""

  new_grid = grid.copy()
  height, width = grid.shape

  for row in range(height):
    for col in range(width):
      if grid[row, col] == 0:
        # Check for horizontal and vertical adjacency to non-white cells
        horizontal_neighbor = (col > 0 and grid[row, col - 1] != 0) or (col < width - 1 and grid[row, col + 1] != 0)
        vertical_neighbor = (row > 0 and grid[row - 1, col] != 0) or (row < height - 1 and grid[row + 1, col] != 0)

        if horizontal_neighbor and vertical_neighbor:
          new_grid[row, col] = 1  # Change to blue

  return new_grid
