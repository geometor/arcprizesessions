import numpy as np

def apply_rule(input_grid):
  """
  Applies the transformation rule to an input grid.

  Args:
    input_grid: A numpy array representing the input grid.

  Returns:
    A numpy array representing the output grid.
  """

  output_grid = input_grid.copy()

  # Find the blue pixel.
  blue_row, blue_col = np.where(input_grid == 1)

  # Create yellow squares around the blue pixel.
  output_grid[blue_row - 2, blue_col - 2] = 4
  output_grid[blue_row + 2, blue_col + 2] = 4
  output_grid[blue_row - 2, blue_col + 2] = 4
  output_grid[blue_row + 2, blue_col - 2] = 4

  return output_grid
