def find_changed_cells(input_grid, output_grid):
    """
    Finds the cells that have changed from input to output and returns their coordinates.
    """
    changed_cells = []
    for row in range(len(input_grid)):
        for col in range(len(input_grid[0])):
            if input_grid[row][col] != output_grid[row][col]:
                changed_cells.append((row, col))
    return changed_cells
