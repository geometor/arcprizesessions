   import numpy as np

   def transform_grid(grid):
       rows, cols = grid.shape
       new_grid = np.copy(grid)
       for i in range(rows):
           for j in range(cols):
               if grid[i, j] == 8:
                   # Check for adjacent white pixels
                   adjacent_white = False
                   for x, y in [(i-1, j), (i+1, j), (i, j-1), (i, j+1)]:
