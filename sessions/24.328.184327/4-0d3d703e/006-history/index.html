import numpy as np

def apply_color_mapping(input_grid, output_grid):
    unique_in = np.unique(input_grid)
    unique_out = np.unique(output_grid)
    color_map = {}
    for i, color in enumerate(unique_in):
        color_map[color] = unique_out[i]
    return color_map

# Example 1
