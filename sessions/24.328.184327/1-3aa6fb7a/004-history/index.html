    visited = set()

def find_adjacent_clockwise(grid, groups, target_color):

        directions = [(r-1, c), (r-1, c+1), (r, c+1), (r+1, c+1), (r+1, c), (r+1, c-1), (r, c-1), (r-1, c-1)]
