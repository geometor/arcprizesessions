import numpy as np

def transform(grid):
    new_grid = grid.copy()
    reds = np.argwhere(grid == 2)
    blues = np.argwhere(grid == 1)

    for r in reds:
        for i in [-2, 2]:
            for j in [-2, 2]:
                if abs(i) != abs(j):
                    y, x = r[0] + i // 2, r[1] + j // 2
                    if 0 <= y < grid.shape[0] and 0 <= x < grid.shape[1]:
                        new_grid[y, x] = 4

    for b in blues:
        for i in [-1, 0, 1]:
            for j in [-1, 0, 1]:
                if abs(i) != abs(j) :
                    y, x = b[0] + i, b[1] + j
                    if 0 <= y < grid.shape[0] and 0 <= x < grid.shape[1]:
