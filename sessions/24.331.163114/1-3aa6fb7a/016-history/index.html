def calculate_intercluster_distances(clusters):
    if not clusters:
        return []
    num_clusters = len(clusters)
    distances = []
    for i in range(num_clusters):
        for j in range(i + 1, num_clusters):
            cluster1_coords = np.array(clusters[i])
