import numpy as np

def transform_grid(input_grid):
    output_grid = np.copy(input_grid)
    rows, cols = input_grid.shape

    for r in range(rows):
        for c in range(cols):
            if input_grid[r, c] == 1:  # Blue pixel
                # Add orange plus sign
                if r > 0: output_grid[r - 1, c] = 7
                if r < rows - 1: output_grid[r + 1, c] = 7
